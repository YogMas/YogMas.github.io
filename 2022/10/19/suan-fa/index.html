# 快速排序

快速排序基于分治的思想，选取数 x (从数组中任选), 将大于 x 的数放置于数组的左边，小于 x 的数放置于数组的右边

处理过程采用双指针和递归进行处理

时间复杂度最快 O (nlogn) 最慢 O (n^2) 取决于每一次随机选取的数在数组中的大小关系

  • 时间复杂度最快

  • 时间复杂度最慢

//快速排序基本模板
void quick_sort(int q[],int l,int r)
{
    if(l >= r) return;
    int x = q[l]; //int x = q[r], int x = q[l+r >>1] 都可以 因为是随机选择
    //这里有个边界问题, 如果后面以j为边界,这里就不能为q[r]  
    //如果后面以i为边界,这里就不能为q[l];
    int i = l -1,j = r + 1; // 先- 先+ 因为后面do while循环 会先进行-- 和++
    while(i < j)
    {
        //比较条件没有等号
        do i++;while(q[i] <x);
        do j--;while(q[j] >x);
        if(i < j) swap(q[i],q[j]);
    }
    quick_sort(q,l,j);
    quick_sort(q,j+1,r);
}

# 归并排序

经典采取分治的思想，时间复杂度为 O (nlogn)

  • 优点：时间复杂度为 O (nlogn);
  • 缺点：使用了多余的空间

通过不断递归，将整个数组无限二等分，直到最后分成单个，在通过比较，将分割后的数组合并起来.

时间复杂度证明可以参照上面的快排最优

//归并排序基本模板
void merge_sort(int q[],int l,int r)
{
    if(l >= r) return;
    int mid = l + r >> 1;
    
    //递归不断二等分
    merge_sort(q,l,mid);
    merge_sort(q,mid + 1,r);
    
    //进行数组合并
    int i =l,j = mid +1,k =0;
    while(i <= mid && j <= r)
    {
        if(q[i] <= q[j]) tmp[k++] =q[i++];                //tmp数组 用空间换时间 临时存储
            else tmp[k++] = q[j++];
    }
    while(i <= mid) tmp[k++] = q[i++];
    while(j <= r) tmp[k++] = q[j++];	
    //将tmp排序好的数组重新赋值到原本的数组中
    for(int i = l,j = 0;i <= r;i++,j++)
    	q[i] = tmp[j];
}

# 整数二分

二分也是采用分治的方法，不断将一个有序数组分成两半，然后判断满足条件的数 (要查找的数) 是在左边还是右边

二分存在边界问题，所以记忆一个模板可以有效避免边界问题。

//当l = r 的时候停止，此时无论是取l 或者 取 r 输出都可以
// 模板1
void find(int q[],int l,int r)
{
    while(l < r)
    {
       	int mid = l + r >>1;
        if(check(mid)) r = mid;  //check(mid)表示是否满足条件
        else l = mid + 1;
    }
}

//模板2
void find(int q[],int l,int r)
{
    while(l < r)
    {
       	int mid = l + r + 1 >> 1
        if(check(mid)) l = mid;  //check(mid)表示中心点是否满足条件
        else r = mid - 1;
    }
}

//通过check函数来判断使用哪个二分
//注意对mid的使用 是导致边界问题的原因

部分问题，查找一个数的值，但数组中有多个一样的值都满足条件，如果确定搜到的答案 是所有满足条件的答案里面最左边的还是最右边的

这个时候需要根据模板 模拟一下， 当条件满足的时候，二分是在往左边查找，还是往右边查找.

  • 往右边查找 最后得到的就是最右边的值
  • 往左边不断查找，最后得到的就是最左边的值
  • 我们可以通过改变 check () 判断函数和模板的切换 来人为控制往左边查找和往右边查找的过程

边界

选择模板为了避免边界问题，灵活记忆

  • 如果满足条件的时候，需要 l = mid , 这个时候求 mid 就是 + 1
  • 如果满足条件的时候，需要 r = mid , 这个时候求 mid 就不需要 + 1, 就是普通的 (l + r) / 2;

# 高精度

C++ 中对大数存储不支持，所以需要用数组来存储大数，然后通过小时候学习的加法竖式等等的原理对数组进行处理，从而达到大数计算的目的

注意：因为进位的问题，所以在加减乘三个运算中反着输入数组，最后再将数组颠倒

  • 因为存在借位问题，减法的最后可能因为被借了一位被判断为 0，所以需要去除这个 0, 去除这个 0 的时候要考虑特例，这个 0 就是答案，所以还要判断是不是只有 1 个 0, 只有满足不止一位数，且最后一位是 0, 才去除这个 0
  • 同样减法也可能存在前置 0, 需要去除
# 高精度加法
//进位问题 输入的数组都是反向存的
vector<int> add(vector<int> &a,vector<int> &b)
{
    vector<int> c;
    for(int i = 0 ,t = 0; t ||i < a.size() || i <b.size();i++)
    {
        if(i <a.size()) t += a[i];
        if(i < b.size()) t += b[i];
        c.push_back(t % 10);
        t /= 10;
    }
    reverse(c.begin(),c.end());
    return c;
}
# 高精度减法
//可以在外面先写个函数 保证输入减函数内部的时候一定是 a >= b的
bool cmp(vector<int> &a,vector<int> &b) //a >= b 则返回true 否则返回false
{
    if(a.size()!= b.size()) return a.size() >b.size(); //长度不一样的时候,显然长度大的更大
    else
    {
        for(int i = 0 ; i < a.size();i++)
        {
            if(a[i] != b[i]) return a[i] >= b[i]; //当某个位置出现不一样的时候,该位置大的更大
        }
    }
    return true; //长度相同 每个位置也相同 说明两个大数相等返回true
}
//默认输入的时候a >= b;
vector<int> sub(vector<int> &a,vector<int> &b)
{
    if(!cpm(a,b)) return sub(b,a); //如果a < b 就进行b - a的操作
    vector<int> c;
    for(int i = 0, t = 0;i < a.size();i++)
    {
        t += a[i];
        if(i <b.size()) t = t - b[i];
        c.push_back((t +10) % 10); //保证插入的是正数
        if(t < 0) t = -1;
        else t = 0;
    }
    while(c.size() >1 && c.back() == 0) c.pop_back(); //删除前置0,如果最后一个为0 并且不只有1个0(正确答案)，就将最后一位山区
    reverse(c.begin(),c.end(0));
    return c;
}

//高精度乘法一般是一个大数和一个小数相乘,不会出现大数乘大数
vector<int> mul(vector<int> &a,int b)
{
    vector<int> c;
    for(int i = 0,t = 0 ; t || i <a.size();i++)
    {
        if(i< a.size())t += a[i] * b;
        c.push_back(t % 10);
        t /= 10;
    }
    reverse(c.begin(),c.end())
    return c;
}
# 高精度除法

高精度除法，不存在借位和进位的问题，所以按照原来的数组存储，但是最后需要翻转两次来去掉前缀 0

//高精度除法,一般也为一个大数和一个小数相除
vector<int> div(vector<int> a,int b,int &r) // r用来存储余数 默认为0
{
    vector<int> c;
    for(int i = 0,r = 0; i< a.size();i++)
    {
        r = r * 10 + a[i];
        c.push_back(r / b); //不够除的时候 自然传入的是0 所以也会有前缀0的问题
        r = r % b; //取余数
    }
    reverse(c.begin(),c.end());
    while(c.size() >1 && c.back() == 0) a.pop_back();
    reverse(c.begin(),c.end());
    return c;
}

# 数学

# 判断质因数

如果一个数除了表示成 1 和他自身的乘积外 还能表示成另外两个数的乘积 这个数就不是质因数

反之一个数如果只能表示成 1 和他自身的乘积，那么这个数是质因数

bool is_prime(int x)
{
	for(int i = 2 ; i <= x /i ;i++) //做了优化  将O(n)时间复杂度降到O(sqrt(n))  这里x/i不写成sqrt(x) sqrt()函数运行较慢 这样每次比较的时候,都会调用sqrt()函数 会大大提高时间
    {
        if(x % i == 0) return false
    }
    return true;
}
# 优化

一个数 表示成两个数的乘积 显然这两个数是一大一小的

如果 x 可以表示成 a * b 的话 假设 a < b 那么显然 a < x / a 即 a < sqrt(x)

# 分解质因数

例如 12 = 2 * 2 * 3 一个合数可以被分解成若干个质数的乘积

分解质因数就是找出一个合数 他所可以分解成的所有质因数

void get_primes(int x)
{
    for(int i = 2; i <= x/ i;i++ )
    {
        if(x % i == 0)
        {
            int s = 0; 
            while(x % i == 0) //将一个质数 判断到底
            {
                x = x / i;
                s++;
            }
            cout << i << s;
        }
    }
    if(x > x / i) cout << x <<endl; //优化 数据单独处理
}
# 优化

分解质因数中 最多只存在一个质数大于 sqrt (x)

所以将最后一个数 脱离出来单独处理 可以将时间复杂度 进行一个降低

前半部分采用遍历的方式 最后一个数据单独处理

# 筛质数

给出一个数 x 把小于等于他的质数全部筛出来

# 埃式筛法 O (nloglogn)

从左往右遍历，找到一个质数的同时，就把 <= n 范围内 他的所有倍数打上标签 (这个数不是质数) 遍历过程中没有打上标签的就是质数

int primes[N]; // 存储所有质数
bool st[N]; //这个数是否被筛过了
int get_primes(int n)
{
    int cnt = 0 ;
    for(int i = 2 ; i <= n;i++)
    {
        if(st[i]) continue; //被筛过了 就跳过
        prime[cnt++] = i;
        for(int j = i + i; j <= n; j += i)   //这里似乎 j 可以直接从i * i 开始 没理解
        {
            st[j] = true;
        }
    }
    return cnt;
}
# 欧式筛法 (线性筛法) O (n)

时间复杂度是线性的

埃式筛法 筛查过程中存在大量的重复筛的问题 比如 2 * 2 = 4 已经被筛过了 但是在到 4 的同时还会被再筛查一遍

所以欧式筛法采用的一个优化的方法 就是筛一个数的时候 只通过最小质因数来筛，这样每个数都只会被筛一遍

bool st[N];
int primes[N];
int get_primes(int n)
{
    int cnt = 0;
    for(int i = 2 ; i <= n;i++)
    {
        if(!st[i]) prime[cnt++] = i;
        for(int j = 0 ; j < cnt && primes[j] <= n \ i;j++)
        {
            st[i * primes[j]] = true;
            if(i % primes[j] == 0) break; //优化所在 如果i是primes[j]的倍数 显然后面 primes[j+1]就不会是最小的质因数了 就break
        }
    }
    return cnt;
}
