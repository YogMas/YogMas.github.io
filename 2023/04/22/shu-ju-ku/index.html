# 子查询

SQL 语句中嵌套 SELECT 语句，称为嵌套查询，又称为子查询

SELECT * FROM WHERE column1 = (SELECT COLUMN1 FROM t2)

# 标量子查询 (子查询结果为单个值)
-- 销售部所有员工信息
select * from emp where emp.dept_id = (select id from dept where name = '销售部')
# 比如此时子查询返回的结果仅仅是单个的id值
-- 查询在方东白入职之后的入职信息
select * from emp where emp.entrydate > (select entrydate from emp where name ='方东白');
# 列子查询 (子查询结果为一列)

常用操作符: IN,NOT IN, ANY, SOME ,ALL

-- 查询销售部和市场部的所有员工信息
#in
select * from emp where emp.dept_id in (select id from dept where dept.name = '销售部' OR dept.name = '市场部')

-- 查询比财务部所有员工工资都高的所有

# all 返回的列表所有都满足条件
select * from emp where emp.salary > all(select salary from emp where dept_id = (select id from dept where name ='财务部'));


-- 查询比研发部其中任意一人工资高的员工信息

#some/any 返回的列表只要有一个满足条件就行
select * from emp where emp.salary > some(select salary from emp where dept_id = (select id from dept where name ='研发部'));
# 行子查询 (子查询结果为一行)

返回的结果是一行 (可以是多列)

常用的操作符: =,<>,IN,NOT IN

-- 查询与张无忌的薪资及直属领导相同的员工信息

select * from emp where (managerid,salary) = (select managerid,salary from emp where name = '张无忌');
# 表子查询 (子查询结果为多行多列)

子查询返回的结果是多行多列

-- 查询与鹿杖客,宋远桥的职位和薪资相同的员工信息

select * from emp where (job,salary) in (select job,salary from emp where name = '鹿杖客' OR name ='宋远桥');

-- 查询入职日期是'2006-01-01' 之后的员工信息,及其部门信息

select * from emp where entrydate > '2006-01-01';

select a.*,b.name from (select * from emp where entrydate > '2006-01-01') as a left join dept as b on a.dept_id = b.id;

# 概述

索引是帮助 mysql 高效获取数据的数据结构 (有序)

# 结构

不加说明的话 说的就是 B+Tree 索引

# B-Tree（多路平衡查找树）

添加数据时，如果超过当前节点的阶数则向上分裂

# B + 树

所有数据都在叶子节点，且叶子节点之间通过指针相互链接， 形成一个单向链表

# Hash

# 分类

# 最左前缀法则

如果索引关系了多列（多个字段), 要遵守最左前缀法则

最左前缀法则：查询从索引的最左列开始 (必须包含最左边的列不包含则索引会全部失效), 并且不跳过索引中的列，若果跳跃某一列，索引将部分失效 (后面的字段索引失效)

# 范围查询

联合索引中，出现范围查询（<,>）, 范围查询右侧的列索引失效

使用 <= >= 则不会失效，在业务允许的情况下 最好都是用 <= 和>=

# 索引失效情况
  • 索引列运算：在索引列上进行运算操作， 索引将会失效
  • 字符串不加引号：字符串类型字段使用时，不加引号，索引将会失效
  • 模糊查询：如果是尾部模糊查询 (xxx%)，索引不会失效， 如果是头部模糊查询，索引失效 (% xxx)
  • OR 连接：用 or 链接的条件如果 or 前的列有索引，or 后的列没有索引，那么设计的索引都不会被用到
  • 数据分布影响：如果 mysql 评估认为走全表比使用索引更快，则不会使用索引
# sql 提示

在 sql 语句中加入一些人为的提示以达到优化的目的 (人为指定数据用或者不用某个索引)

  • 使用某个索引 (use index): explain select * from tb_user use indx(idx_user_pro) where profession = '软件工程' (只是给 mysql 使用这个索引的建议 注意与强制索引区分)
  • 不适用某个索引 (ignore index): explain select * from tb_user ignore indx(idx_user_pro) where profession = '软件工程'
  • 强制使用某个索引 (force index)： explain select * from tb_user force indx(idx_user_pro) where profession = '软件工程'
# 覆盖索引

尽量使用覆盖索引 (查询使用索引，并且返回的列在该索引中已经全部能够找到), 减少 select *

建立 username 和 password 的联合索引是最佳的，因为不需要回表查询

# 前缀索引

可以只将字符串的一部分前缀建立索引，大大节约索引空间，从而提高索引效率

create index idx_xxx on 表名(列名(n)) : 表示提取字符串前 n 个字符来建立索引

前缀长度

前缀长度可以根据索引的选择性来决定， 选择性是指不重复的索引值和数据表的记录总数的比值，选择性越高则查询效率越高

select count(distinct email)/count(*) from tb_user;

select count(distinct substring(email,1,xxx))/count(*) from tb_user;

# 单列 & 联合索引

涉及多个条件时，建议使用联合索引，而非单列索引，因为这样可以减少回表查询

多条件联合查询时，mysql 会评估使用哪个字段的索引效率更高，会使用该索引完成本次查询

# limit 优化

limit 越往后查询，性能越低

比如 limit 查询 200000~200010 的记录，mysql 需要对前 200010 条记录进行排序，却只返回前 10 条记录，其他记录全部丢弃，所以查询排序的代码非常大

优化思路：使用覆盖查询 + 子查询的形式，查询 id 时使用覆盖查询，就可以减少前面的数据进行回表查询，在通过得到的 ID 进行回表查询，此时进行回表查询的数量仅仅是我们需要得到的数据的数量，效率大大提升

innoDb 中行锁是根据索引加的锁

执行 update 字段，要根据索引进行更新，否则行锁会升级成为表锁

# 视图

视图是虚拟存在的表，视图中的数据并不在数据库中实际存在，数据来自自定义视图查询中所用的表，并且是在使用视图时动态生成的

视图只保存了查询的 sql 逻辑，不保存查询结果

  • 创建视图: CREATE [OR REPLACE] VIEW 视图名称[(列表名)] AS SELECT语句 [WITH [CASCADED | LOCAL] CHECK OPTION]
  • 查询视图
    1. 查询创建视图语句: show create view 视图名称
    2. 查看视图数据: SELECT * FROM 视图名称 ....;
  • 修改视图
    1. 把对应视图内容替换: CREATE [OR REPLACE] VIEW 视图名称[(列表名)] AS SELECT语句 [WITH [CASCADED | LOCAL] CHECK OPTION]
    2. 替换视图: ALTER VIEW 视图名称[(列表名)] AS SELECT语句 [WITH [CASCADED | LOCAL] CHECK OPTION]
  • 删除: DROP VIEW [IF EXISTS] 视图名称[,视图名称]..

可以向视图中插入数据，但视图中不存储数据，所以向视图中插入数据时，其实是向之前的来源表中插入数据

但是可能存在插入数据不满足视图的查询条件，虽然插入到原来的表中了，但是视图却获取不到，所以下面提供了 2 个查询选项

检查选项 with check option 默认值为 CASCADED

  1. 检查选项 CASCADED

    加上此检查选项后，数据库会检查所有层联的视图，看是否满足查询条件

  2. 检查选项 Local

    他也会递归所有层联的视图，但是和 CASCADED 的区别是如果层联的视图没有 with check option 他则不会检查，但是 CASCADED 会检查

# 存储过程

存储过程是实现经过编译并存储在数据库中的一段 sql 语句的集合

存储过程就是数据库 sql 语句的代码封装和重用

特点

  • 封装，复用
  • 可以接收参数，也可以返回数据
  • 减少网络交互，效率提升

# loop

实现简单的循环，如果不增加退出循环的条件，就会实验简单的死循环

和汇编的 loop 类似

  • leave : 配合循环使用，退出循环
  • iterate : 必须使用在循环中，作用跟 continue 相同
-- a啊
[begin_label:]Loop

end loop;[end_label];
   
 leave label; #退出循环
 iterate label; #进入下一个循环
#例子
-- leave
create procedure p9(in n int)
begin
    declare result int default 0;
    begin_sum:loop
        if n <= 0 then leave begin_sum;
        else
        set result = result + n;
        set n = n -1;
        end if;
        end loop begin_sum;
    select result;
end;

call p9(10);


-- iterate
create procedure p10(in n int)
begin
    declare result int default 0;
    total:loop
    if n % 2 = 1 then
        set n = n -1;
        iterate total;
    elseif n <= 0 then leave total;
    else
        set result = result + n;
        set n = n -1;
    end if;
    end loop total;
    select result;
end;
call p10(10);

# 游标

存储查询结果集的数据类型，在存储过程和函数中可以使用游标对结果集进行循环的处理

游标的使用包括游标的使用，open,fetch 和 close

  • 声明游标： declare 游标名称 cursor for 查询语句
  • 打开游标: open 游标名称
  • 获取游标记录: fetch 游标名称 into 变量[,变量]

声明时， 需要先声明普通变量再生成游标

#游标的使用
create procedure p10(in uage int)
begin
    declare uname varchar(50);
    declare upro varchar(50);
    declare user_cur cursor for select name,profession from tb_user where age <= uage;
    create table new_user(
        id int primary key auto_increment comment '编号' ,
        name varchar(50) comment '姓名',
        profession varchar(50) comment '职业'
    )comment '新表';
    open user_cur;
    while true do
        fetch user_cur into uname,upro;
        insert into new_user (name,profession) values (uname,upro);
    end while;
    close user_cur;
end;

# 条件处理程序

在流程控制结构执行过程中遇到问题时相应的处理步骤

declare handler_action HANDLER FOR condition_value statement(语句)

-- 参数
handler_action:
	continue:继续执行
	exit:终止

condition_value
	sqlstate 状态码
	sqlwarning 以01开头的sqlstate代码的简写
	not found:以02开头的sqlstate代码的简写
	sqlexception: 所有没有被sqlwarning 和not found捕获的sqlstate代码的简写

create procedure p12(in uage int)
begin
    declare uname varchar(50);
    declare upro varchar(50);

    declare exit handler for sqlstate '02000' close user_cur;

        name varchar(50) comment '姓名',
        profession varchar(50) comment '职业'
    )comment '新表';
    open user_cur;
    while true do
        fetch user_cur into uname,upro;
        insert into new_user (name,profession) values (uname,upro);
    end while;
    close user_cur;
end;

# 存储函数

存储函数是有返回值的存储过程，存储函数的参数只能是 in 类型，且必须有返回值

create function 存储函数名称([参数列表])
returns 返回类型[characteristic ...]
begin
	-- sql语句
	return...;
end;
characteristic说明:
- deterministic:相同的输入总是产生相同的结果
- No sql:不包含sql语句
- reads sql data: 包含读取数据的语句但不包含写入数据的语句
create function fun1(n int) #默认就是in所以不用写
returns int deterministic
begin
    declare result int default 0;
    repeat
        set result := result + n;
        set n := n -1;
    until  n <= 0
    end repeat;
    return result;
end;

select fun1(50);

存储函数使用相对较少，因为存储函数能实现的存储过程都能实现

# 触发器

指在 insert/update/delete 之前或之后，触发并执行触发器中定义的 sql 语句集合

使用别名 OLD 和 NEW 来引用触发器中发生变化的内容

目前触发器只支持行级触发，不支持语句级触发

  • 行级触发：改变了多少行，触发多少次
  • 语句级触发：一条语句触发一次

# 语法

# 创建
create trigger trigger_name
BEFORE/AFTER INSERT/UPDATE/DELETE
ON tbl_name for each row -- 行级触发器
BEGIN
	trigger_stmt;
end;
# 查看
show triggers;
#删除
drop trigger[schema_name]trigger_name # 没有schema_name 则默认为当前数据库
#例子

-- insert触发器
create trigger tb_user_insert_trigger
after insert on tb_user for each row
begin
    insert into user_logs(operation, operate_time, operate_id, operate_params) values('insert',now(),new.id,concat('插入的数据内容为 id=',new.id,'name=',new.name,'email=',new.email,'profession=',new.profession));  # 当insert之后会把日志插入日志表
end;

-- update触发器
create trigger tb_user_update_trigger
after update on tb_user for each row
begin
insert into user_logs(operation, operate_time, operate_id, operate_params) values('update',now(),new.id,concat('更新之前的数据为 id=',old.id,'name=',old.name,'email=',old.email,'profession=',old.profession));
end;

-- delete触发器
create trigger tb_user_delete_trigger
after delete on tb_user for each row
begin
    insert into user_logs(operation, operate_time, operate_id, operate_params) values('delete',now(),old.id,concat('删除的数据为 id=',old.id,'name=',old.name,'email=',old.email,'profession=',old.profession));
end;

# 锁

  • 全局锁：锁定数据库所有表
  • 表级锁：每次操作锁住整张表
  • 行级锁：每次操作锁住对应的行数据

# 全局锁

对整个数据库加锁，加锁后整个实例就处于只读状态，后续的写语句事务提交语句都将被阻塞

典型的使用场景是做全库的逻辑备份，对所有的表进行锁定，从而获取一致性视图，保证数据的完整性

  • 加全局锁: flush tables with read lock;
  • 备份: mysqldump -u root -p123 itcast>itcast.sql;
  • 解锁: unlock tables;

# 表级锁

锁住整张表，发生锁冲突的概率最高，并发度最低

主要分为三类

  • 表锁
  • 元数据锁
  • 意向锁
# 表锁
  1. 表共享读锁: 所有客户端只能读不能写
  2. 表独占写锁: 本客户端既能读又能写，但是其他客户端不能读也不能写
  • 加锁: lock tables 表名 read/write
  • 释放: unlock tables / 客户端断开连接

# 元数据锁 (meta data lock, MDL)

查看元数据锁： select object_type,object_schema,object_name,lock_type,lock_duration from performance_schema.metadata_locks;

# 意向锁

  • 意向共享锁 (IS)： select ... lock in share mode 与表锁共享锁 (read) 兼容，与表锁排他锁 (write) 互斥
  • 意向排他锁 (IX)： insert,update,delete,select ... for update 添加 与表锁共享锁 (read) 和表锁排他锁 (write) 都互斥 意向锁之间不会互斥

查看意向锁及行锁的加锁情况： select OBJECT_SCHEMA,OBJECT_NAME,INDEX_NAME,LOCK_TYPE,LOCK_MODE,LOCK_DATA from performance_schema.data_locks

# 行级锁

每次操作锁住对应的行数据，发生锁冲突的概率最低，并发度最高，应用在 innoDB 引擎中

  • 行锁：锁定单个记录的锁，防止其他事务对此进行 update 和 delete, 在 RC,RR 隔离级别下都支持
  • 间隙锁：锁定索引记录间隙，确保索引记录间隙不变，防止其他事务在这个间隙进行 insert, 产生幻读，在 RR 隔离级别下都支持
  • 临键锁：行锁和间隙锁的组合，同时锁住数据，并锁住数据前面的间隙 Gap, 在 RR 隔离级别下支持
